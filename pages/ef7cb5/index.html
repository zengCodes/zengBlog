# 对象或者数组深度

function maxLevelNumber(obj = {}) {
     let level = 1
     const fun = (item, lev = 0) => {
        if (typeof item === 'object' && item != null) {
            Object.values(item).forEach(element => {
              if (typeof element === 'object' && element != null) {
                  fun(element, lev + 1)
               } else {
                  level = Math.max(lev + 1, level)
               }
             });
         } else {
            level = Math.max(lev, level)
         }
     }
   fun(obj)
   return level
}
// 最大对象层数
let object = {
    a: 0,
    b: 1,
    c: {
      a: 0
    },
    d: {
    a: {
       a: 0
    }
   }
}
console.log(maxLevelNumber(object)); // 3

function add(args) {
    return args.reduce((pre, cur) => {
       return pre+cur
   },0)
}
const foo = (...args1) => {
    // 保存第一次求和
    let sum1 = add(args1)
    const fun = (...args2) => {
        let sum2 = add(args2)
        return foo(sum1+sum2)
    }
    fun.toString = () => {
        return sum1
    }
    return fun
}
console.log(foo(1, 2, 3) == 6);
console.log(foo(1, 2)(3) == 6);
console.log(foo(1)(2,3) == 6);

# 排序算法

  • 冒泡排序
    • 对相邻元素进行两两比较，顺序不一致则交换位置，每次比较都会将最大或者最小的元素置于顶端
  • 选择排序
    • 每次循环，选择一个最小或者最大的元素置于前面的有序序列中末尾处
  • 插入排序
    • 每次循环，将元素插入到左侧有序序列中
    • 插入过程中，从有序序列右侧开始比较，若小于序列中的树，则下标右移
  • 快速排序
    • 选择一个基准，把小的数和大的数放在两个序列中，利用递归把两个序列重复操作，进行合并
  • 归并排序
    • 利用分治思想，先把子序列进行排序，组成一个有序子序列，再将多个有序子序列合并成一个有序序列
  • 希尔排序
    • 将元素按下标一定增量进行分组，再利用插入排序进行排序，
  • 堆排序
    • 堆分为最大堆和最小堆，最大堆中根节点元素大于其孩子节点元素，最小堆根节点小于其孩子节点元素
    • 选择排序的一种，将最大堆中堆顶最大元素取出，将剩余的堆调整成最大堆，继续去除堆顶最大元素，直到堆内只剩一个元素为止

# 树

  • 树：根节点、左右子树、深度
  • 二叉树
    • 根节点最多只有两个子节点
  • 二叉搜索树
    • 根节点左侧节点只存储小（比右节点）的节点
    • 最小节点在最左边，最大节点在最右边
  • 满二叉树
    • 除最后一层没有子节点外，每层的节点都有左右两个节点
  • 完全二叉树
    • 除最后一层外，其根节点都有左右节点，并且最后一层的节点尽可能在左侧
  • 平衡二叉树
  • 红黑树
  • 堆
class node {
    constructor(val) {
        this.key = val; //  节点值
        this.left = null; // 左侧子节点引用
        this.right = null; // 右侧子节点引用
    }
}
class binaryTree {
    constructor() {
        this.root = null
    }
    // 搜索方法
    // 搜索最小值
    min() {
        return this.minNode(this.root);
    }
    // 搜索最大值
    max() {
        return this.maxNode(this.root);
    }
    // 最小值辅助方法
    minNode(node) {
        let current = node;
        while (current != null && current.left != null) {
            current = current.left;
        }
        return current;
    }
    // 最大值辅助方法
    maxNode(node) {
        let current = node;
        while (current != null && current.right != null) {
            current = current.right;
        }
        return current;
    }
    // 搜索节点是否存在
    search(key) {
        return this.searchNode(this.root, key);
    }
    // 插入节点
    insert(val) {
        let newNode = new node(val)
        if (this.root === null) {
            this.root = newNode
        } else {
            this.insertNode(this.root, newNode)
        }
    }
    // 删除节点
    remove(key) {
        this.root = this.removeNode(this.root, key);
    }
    // 先序遍历
    preOrderTraverse(callback) {
        this.preOrderTraverseNode(this.root, callback);
    }
    // 中序遍历
    inOrderTraverse(callback) {
        this.inOrderTraverseNode(this.root, callback);
    }
    // 后序遍历
    postOrderTraverse(callback) {
        this.postOrderTraverseNode(this.root, callback);
    }
    // 搜索节点辅助方法
    searchNode(node, key) {
        if (node === null) {
            return false
        }
        if (key < node.key) {
            return searchNode(node.left, key)
        } else if (key > node.key) {
            return searchNode(node.right, key)
        } else {
            return true
        }
    }
    // 插入辅助方法
    insertNode(root, curNode) {
        if (root.key > curNode.key) {
            if (root.left === null) {
                root.left = curNode
            } else {
                this.insertNode(root.left, curNode)
            }
        } else {
            if (root.right === null) {
                root.right = curNode
            } else {
                this.insertNode(root.right, curNode)
            }
        }
    }
    removeNode(node, key) {
        if (node == null) {
            return null;
        }
        if (key < node.key) {
            node.left = removeNode(node.left, key);
            return node
        } else if (key > node.key) {
            node.right = removeNode(node.right, key);
            return node
        } else {
            // 键等于 node.key 
            //  第一种情况 一个根节点 
            if (node.left === null && node.right === null) {
                node = null;
                return node
            }
            // 第二种情况 只有一个子节点的节点
            if (node.left === null) {
                node = node.right;
                return node
            } else if (node.right === null) {
                node = node.left;
                return node
            }
            // 第三种情况 有两个子节点的节点
            const aux = this.minNode(node.right);
            node.key = aux.key;
            node.right = this.removeNode(node.right, aux.key);
            return node;
        }
    }
}

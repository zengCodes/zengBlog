# 数组中的length

  • 数组的length是可以修改的
  • 数组删除元素后，元素下标还存在，元素为undefined
let arr = [1, 2, 3, 4, 5, 6]
console.log(arr.length); // 6
delete arr[1]
console.log(arr); // [1, 空白, 3, 4, 5, 6]
console.log(arr.length); // 6
arr.length = 2
console.log(arr); // [1,undefined]
console.log(arr.length); // 2
arr[10] = 10
console.log(arr); // [1, 空属性 × 9, 10]
let newArr = arr.filter(item => {
    return item == undefined
})
console.log(newArr.length); // 0

# 转换规则

  • To Primitive：对象需要先转换为基本数据类型
    • 先调用valueof()方法
    • 如果没有valueof()方法或者返回的值不是基本数据类型，就继续调用toString()方法
    • 如果没有toString()方法或者返回的的值不是基本数据类型，就抛出TypeError错误
 console.log([1, null, NaN, undefined, '', '0',false, true].toString()); // 1,,NaN,,,0,false,true
 console.log([1, Symbol('age')].toString()); // TypeError: Cannot convert a Symbol va    lue to a string
 console.log(Object.create({}).toString()); // [object Object]
 console.log(Object.create(null).toString()); // Object.create(...).toString is not a function
 console.log(Object.create(undefined).toString()); // Object prototype may only be an Object or null: undefined

# To Number：将值转数字

  • undefined => NaN
  • null => +0
  • boolean => false: 0, true: 1
  • string => '' : 0, 有字符串数字解析为数字，e.g. '12.3': 12.3,'xxx': NaN
  • number => number 无需转换
  • object => 先ToPrimivetive(obj, preferredType=Number)，然后再进行ToNumber
  • 将不能转数字的转NaN
console.log(Number(null)); // 0 
console.log(Number(undefined)); // NaN
console.log(Number(true)); // 1
console.log(Number(false)); // 0
console.log(Number('')); // 0
console.log(Number('0')); // 0
console.log(Number('1')); // 1
console.log(Number('1.1')); // 1.1
console.log(Number('{}')); // NaN
console.log(Number('[]')); // NaN
console.log(Number([])); // 0
console.log(Number({})); // NaN
console.log(Number('x')); // NaN
console.log(Number([1, 2, 3])); // NaN

# 运算符

  • 加法
    • 如果两个操作数是字符串，直接拼接字符串
    • 如果只有一个操作数是字符串，那么将另外一个操作数转字符串，在拼接字符串
    • 如果一个操作数是对象、布尔值、数字，则调用toString方法，在计算

console.log([] + 1); // 1

# 总结

# 类型相同

  • 不需要类型转换
  • NaN与任意类型不等

# 类型不同

  • a、b 为null和undefined两者中的一个，则返回true
  • a、b为Number和String类型时，转换为Number进行比较
  • 若Boolean类型时，转化为Number比较
  • 一个Object类型，一个String或者Number类型，将Object按类型转换为原始值后，再按上面三步比较

# 浅拷贝和深拷贝

  • 数据类型存储
    • 基本数据类型保存在栈内存中
    • 引用数据类型保存在堆内存中，应用数据类型的变量是一个指向堆内存中实际对象的引用，存在栈中
  • 浅拷贝
    • 对于基本数据类型，只拷贝值，如果是引用数据类型，则拷贝引用地址
    • 浅拷贝只拷贝一层，深层次的引用类型共享引用地址
  • 深拷贝
    • 开辟新的栈，两个对象属性相同，但是对应不同的地址，修改拷贝对象，不会影响原来对象
  • 区别
// Object.assign和赋值实现浅拷贝
let a = {
 name: "tom",
 user: {
    age: 10
 }
}
let b = {}
Object.assign(b, a)
// b = a
b.name = "vat"
b.user.age = 11
console.log(a); // 第一层是原始数据不会改变，第一层不是基本数据会改变
// Array.prototype.slice实现浅拷贝
let arr1 = [1, 2, 3, 4]
let newArr1 = arr1.slice()
arr1[1] = 5
console.log(arr1); // [1, 5, 3, 4] 改变原来数据
// Array.prototype.concat实现浅拷贝
let arr2 = [1, 2, 3, 4]
let newArr2 = arr2.concat()
arr2[1] = 5
console.log(arr2); // [1, 5, 3, 4] 改变原来数据

// JSON.stringify()实现深拷贝
let c = {
   name: "tom",
   user: {
      age: 10
    }
}
let d = {}
d = JSON.parse(JSON.stringify(c))
d.name = "vat"
d.user.age = 11
console.log(c);

// 扩展运算符
let f = {
    name: "tom",
    user: {
        age: 10
    }
 }
let g = {
 ...f
}
g.name = "vat"
g.user.age = 11
console.log(f); // 第一层是原始数据不会改变，第一层不是基本数据会改变

# this指向问题

  • 绑定规则
    • 默认绑定规则
    • 隐式规则
    • new 绑定
    • 显示规则
// 默认指向window
var name = "tom"
function User() {
   console.log(this.name);
}
User() // tom

// 隐式绑定指向调用对象
let dog = {
    name: '小哈',
    say: function () {
        console.log(this.name);
    },
    call: {
        say: function () {
            console.log(this.name);
        }
    }
}
dog.say() // 小哈
dog.call.say() // undefined
var fun = dog.call.say
fun() // tom

// new绑定
function Person() {
    this.name = "小王"
}
let p = new Person()
console.log(p.name); // 小王

// 显示绑定 通过apply、call、bind指定this指向
let obj = {
  food: '西瓜',
  callback: {
    food: '玉米',
    eat: () => {
        console.log(this.food);
     }
  },
  eat: function () {
        console.log(this.food);
    }
}
obj.eat() // 西瓜
let oldFun = obj.eat
oldFun() // undefined
oldFun.call(obj.callback) // 玉米
obj.eat.call(this) // undefined
obj.eat.call(obj.callback) // 玉米
// 箭头函数没有this指向,所以定义callback中的eat方法时，会默认绑定到window
obj.callback.eat() // undefined
obj.callback.eat.call(obj) // undefined
let newFun = obj.callback.eat
newFun() // undefined

# 随机数

  • Math.radom返回一个[0,1)的数
  • Math.round返回一个四舍五入最接近的整数
  • Math.floor返回一个小于或者等于给定数字的最大整数
  • Math.ceil返回大于或者等于给定数字的最大整数

# min <= r <= max

function random(min, max) {
    return Math.round(Math.random() * (max - min)) + min;
}

# 闭包

  • 定义：一个可以访问其外部函数的作用域的函数
  • 优点：避免全局变量的使用、延长变量生命周期
  • 缺点：增加内存使用，有可能造成内存泄露
  • 解决办法：退出函数时，清除变量
  • 使用场景：使用闭包实现函数柯里化，防抖，节流，延迟回调，计数器等

# 内存泄漏

  • 循环引用
  • 闭包
  • 定时器开启后，在组件卸载时没有清除
  • 全局变量的产生
    • 严格模式下可以避免
  • 没有清除dom元素引用
  • dom元素删除后，事件监听没有删除

# 作用域

  • 最小暴露原则

// 实例1
let bb = 1
function fun(arg) {
  bb = 2
  console.log(bb); // 2
}
fun(bb)
console.log(bb); // 2


// 实例2
var aa = 1
function fun() {
  console.log(aa); // err 不能在未定义之前使用
  let aa = 0
  bar()
  console.log(aa); // 0 
}
function bar() {
  console.log(aa); // 1
}
fun();
